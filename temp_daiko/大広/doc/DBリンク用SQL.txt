CREATE TABLE DPTPOST
(
    POST_ID                        VARCHAR2(40) NOT NULL,
    POST_CODE                      VARCHAR2(30) NOT NULL,
    POST_NAME                      VARCHAR2(40),
    BASE_ID                        VARCHAR2(40),
    COMPANY_ID                     VARCHAR2(40 CHAR) NOT NULL,
    CONSTRAINT DPTPOST_PK PRIMARY KEY (POST_ID) USING INDEX ENABLE
)
;
COMMENT ON TABLE DPTPOST IS '役職'
;
COMMENT ON COLUMN DPTPOST.POST_ID IS '役職ID'
;
COMMENT ON COLUMN DPTPOST.POST_CODE IS '役職コード'
;
COMMENT ON COLUMN DPTPOST.POST_NAME IS '役職名称'
;
COMMENT ON COLUMN DPTPOST.BASE_ID IS '拠点ID'
;
COMMENT ON COLUMN DPTPOST.COMPANY_ID IS '法人ID'
;


CREATE TABLE PERSON
(
    PERSON_ID                      VARCHAR2(40) NOT NULL,
    PERSON_CODE                    VARCHAR2(30) NOT NULL,
    PERSON_NAME                    VARCHAR2(40),
    HIRED_DATE                     CHAR(10),
    WORKING_START_DATE             CHAR(10),
    RETIRE_DATE                    CHAR(10),
    MAIL_ADDRESS                   VARCHAR2(255),
    COMMENT_TEXT                   VARCHAR2(255),
    ATTR_DECIDE_DATE               CHAR(10 CHAR),
    FUNCTION_GROUP_CODE            VARCHAR2(30 CHAR),
    ROLE_ID                        VARCHAR2(40 CHAR),
    COMPANY_ID                     VARCHAR2(40 CHAR) NOT NULL,
    CONSTRAINT PERSON_PK PRIMARY KEY (PERSON_ID) USING INDEX ENABLE
)
;
COMMENT ON TABLE PERSON IS '人'
;
COMMENT ON COLUMN PERSON.PERSON_ID IS '人ID'
;
COMMENT ON COLUMN PERSON.PERSON_CODE IS '人コード'
;
COMMENT ON COLUMN PERSON.PERSON_NAME IS '氏名'
;
COMMENT ON COLUMN PERSON.HIRED_DATE IS '入社年月日'
;
COMMENT ON COLUMN PERSON.WORKING_START_DATE IS '運用開始日'
;
COMMENT ON COLUMN PERSON.RETIRE_DATE IS '運用終了日'
;
COMMENT ON COLUMN PERSON.MAIL_ADDRESS IS 'メールアドレス'
;
COMMENT ON COLUMN PERSON.COMMENT_TEXT IS 'コメント'
;
COMMENT ON COLUMN PERSON.ATTR_DECIDE_DATE IS '属性決定日付'
;
COMMENT ON COLUMN PERSON.FUNCTION_GROUP_CODE IS '機能グループコード'
;
COMMENT ON COLUMN PERSON.ROLE_ID IS 'ロールID'
;
COMMENT ON COLUMN PERSON.COMPANY_ID IS '法人ID'
;




CREATE TABLE PARTY_RELATION
(
    PARTY_ID                       VARCHAR2(40) NOT NULL,
    PARTY_RELATION_TYPE            VARCHAR2(30) NOT NULL,
    MASTER_ID                      VARCHAR2(40) NOT NULL,
    PERIOD_START                   CHAR(10) NOT NULL,
    PERIOD_END                     CHAR(10),
    CONSTRAINT PTREL_PK PRIMARY KEY (PARTY_RELATION_TYPE, MASTER_ID, PARTY_ID, PERIOD_START) USING INDEX ENABLE
);

COMMENT ON TABLE PARTY_RELATION IS 'パーティ関連'
;
COMMENT ON COLUMN PARTY_RELATION.PARTY_ID IS 'パーティID'
;
COMMENT ON COLUMN PARTY_RELATION.PARTY_RELATION_TYPE IS 'パーティ関連型'
;
COMMENT ON COLUMN PARTY_RELATION.MASTER_ID IS 'マスタID'
;
COMMENT ON COLUMN PARTY_RELATION.PERIOD_START IS '有効期間開始日'
;
COMMENT ON COLUMN PARTY_RELATION.PERIOD_END IS '有効期間終了日'
;
